	INNER JOIN CURRENCY2COUNTRY cc ON p.CURRENCY_ID = cc.CURRENCY_ID

CREATE OR REPLACE FORCE EDITIONABLE VIEW "TAXI_SHAPOSHNIKOV"."TOP5ADDRESS" ("PASSENGER_NAME", "ADDR") AS 
  SELECT p.PASSENGER_NAME, t.addr FROM (
SELECT io.PASSENGER_ID, w.TO_ADDRESS_ID AS addr, count (w.TO_ADDRESS_ID) AS cnt   FROM IS_ORDER io
left OUTER  JOIN way w ON w.ORDER_ID = io.ID 
GROUP BY io.PASSENGER_ID, w.TO_ADDRESS_ID
HAVING count (w.TO_ADDRESS_ID) > 1
ORDER BY 1 ASC) t
JOIN PASSENGER p ON p.ID = t.PASSENGER_ID
GROUP BY p.PASSENGER_NAME, t.addr
ORDER BY 1,2 ASC;
;

CREATE OR REPLACE PROCEDURE TAXI_SHAPOSHNIKOV.get_mile_cost(
addrVarray IN varchar, -- получаю точки адреса (остановки)
mileage OUT float, -- отдаю расстояние
cost OUT float -- отдаю стоимость
)
AS 

names addr_points;
counter NUMBER;
x varchar(200);
total NUMBER;
mileage_a float;

mileage_cost_a float;

BEGIN
mileage_cost_a:=0.0;
names := addr_points();
counter:=regexp_count(addrVarray,'[^:]+',1);

		FOR i in 1 .. counter LOOP
		
			x:=regexp_substr(addrVarray, '[^:]+',1,i);
			names.extend;
			names(i) := x;

		total:=names.count;
	
		SELECT * INTO mileage_a FROM (SELECT (c.id+c2.id+s.id+a.id)
									FROM COUNTRY c 
										JOIN CITY c2 ON c.ID =c2.COUNTRY_ID 
										JOIN STREET s ON s.CITY_ID = c2.ID	
										JOIN ADDRESS a ON a.STREET_ID = s.ID
										WHERE a.id = names(i)); -- считаю расстояния
					mileage_cost_a:=mileage_cost_a+mileage_a;
					mileage_a:=mileage_a+mileage_a;
		END LOOP;
	
	cost:=mileage_cost_a/10;
	mileage:=mileage_a/100;

END;

       driver number;

                        BEGIN
                        IF r.is_reserved = 0 THEN RAISE ISRESERVED;
----------------------------------------------------------------------
                        ELSE

                                                R.GAS_MILEAGE = FUEL_OUT,

                                                WHERE R.CAR_ID = CARID
                                                AND r.ID = rent_id;

                            END;
                  END LOOP;
END;
